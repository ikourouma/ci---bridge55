    <script>
        // 1. Time Logic
        function updateTime() {
            const now = new Date();
            document.getElementById('localTime').textContent = now.toLocaleTimeString('en-GB', {timeZone: 'Africa/Abidjan', hour:'2-digit', minute:'2-digit'});
        }
        setInterval(updateTime, 60000); updateTime();

        // 2. User State Logic (Welcome Text)
        let isLoggedIn = false;
        const userName = "Jean-Philippe";
        
        function toggleUserState() {
            isLoggedIn = !isLoggedIn;
            const subText = document.getElementById('logoSubText');
            const dropWelcome = document.getElementById('dropdownWelcome');
            const btn = document.querySelector('.btn-test');
            
            if (isLoggedIn) {
                // Show Welcome Message
                subText.innerHTML = `<span class="welcome-user">Welcome, ${userName}</span>`;
                dropWelcome.textContent = `Hi, ${userName}`;
                btn.textContent = "Simulate Logout";
                btn.style.backgroundColor = "#EF4444";
            } else {
                // Reset
                subText.textContent = "visit & explore";
                dropWelcome.textContent = "Welcome, Guest!";
                btn.textContent = `Simulate Login: ${userName}`;
                btn.style.backgroundColor = "";
            }
        }

        // 3. Toggles
        function toggleProfile(e) { e.stopPropagation(); document.getElementById('profileDropdown').classList.toggle('active'); }
        function toggleMobileMenu() { document.getElementById('mobileMenu').classList.toggle('open'); }
        function toggleWeatherPopup(e) { e.stopPropagation(); document.getElementById('weatherPopup').classList.toggle('active'); }
        function openDia() { alert('Dia AI Concierge launching...'); }
        
        // 4. Mega Menu Interaction
        const overlay = document.querySelector('.mega-menu-overlay');
        const navItems = document.querySelectorAll('.nav-item.has-mega');
        
        navItems.forEach(item => {
            item.addEventListener('mouseenter', () => {
                const id = item.dataset.mega;
                document.querySelectorAll('.mega-menu').forEach(m => m.classList.remove('open'));
                navItems.forEach(n => n.classList.remove('active'));
                
                const menu = document.getElementById('mega-' + id);
                if (menu) {
                    menu.classList.add('open');
                    item.classList.add('active');
                    overlay.classList.add('visible');
                }
            });
        });

        // 5. Dynamic Images Logic
        document.querySelectorAll('.mega-link').forEach(link => {
            link.addEventListener('mouseenter', function() {
                const container = this.closest('.mega-grid');
                if (!container) return;
                
                // Active state
                container.querySelectorAll('.mega-link').forEach(l => l.classList.remove('active'));
                this.classList.add('active');
                
                // Elements
                const img = container.querySelector('.mega-hero-img');
                const title = container.querySelector('.mega-hero-title');
                const desc = container.querySelector('.mega-hero-desc');
                const badge = container.querySelector('.mega-hero-badge');
                
                // Swap data
                if (img && this.dataset.img) {
                    img.classList.add('fade-out');
                    setTimeout(() => {
                        img.src = this.dataset.img;
                        img.classList.remove('fade-out');
                    }, 200);
                }
                if (title && this.dataset.title) title.textContent = this.dataset.title;
                if (desc && this.dataset.desc) desc.textContent = this.dataset.desc;
                if (badge && this.dataset.badge) badge.textContent = this.dataset.badge;
            });
        });

        // 6. Global Close
        document.addEventListener('click', (e) => {
            if (!e.target.closest('.nav-item') && !e.target.closest('.mega-menu')) {
                document.querySelectorAll('.mega-menu').forEach(m => m.classList.remove('open'));
                navItems.forEach(n => n.classList.remove('active'));
                overlay.classList.remove('visible');
            }
            if (!e.target.closest('.profile-wrapper')) {
                document.getElementById('profileDropdown').classList.remove('active');
            }
            if (!e.target.closest('#weatherWidget')) {
                document.getElementById('weatherPopup').classList.remove('active');
            }
        });
    </script>
